\begin{tikzpicture}[every node/.style={circle,draw,minimum size=2em}, >=latex,
    myshadow/.style={
        blur shadow={shadow blur steps=5, shadow xshift=0.15em, shadow yshift=-0.1em}
    }]

    \def\Km{5}
    \def\Kn{4}
    \def\Kmskip{4}
    \def\Knskip{3}
    \def\voff{-0.575}
    \def\dotsoff{0.15}
    
    \def\UVsep{2.8}
    \definecolor{verylightgray}{rgb}{0.9, 0.9, 0.9}
    \colorlet{verylightblue}{cyan!20}
    \colorlet{verylightyellow}{yellow!20}
    
    \foreach \i in {1,...,\Km} {
        \ifnum\i=\Kmskip
          \node[draw=none] (a\i) at (0,\dotsoff-\i) {\LARGE $\vdots$};
        \else
            \ifnum\i=\Km
                \node[draw=orange,fill=verylightyellow,myshadow] (a\i) at (0,-\i) {$u_m$};
            \else
                \node[draw=orange,fill=verylightyellow,myshadow] (a\i) at (0,-\i) {$u_{\i}$};
            \fi
        \fi
    }
    \foreach \i in {1,...,\Kn} {
        \ifnum\i=\Knskip
            \node[draw=none] (b\i) at (\UVsep,\dotsoff+\voff-\i) {\LARGE $\vdots$};
        \else
            \ifnum\i=\Kn
                \node[draw=blue,fill=verylightblue,myshadow] (b\i) at (\UVsep,\voff-\i) {$v_n$};
            \else
                \node[draw=blue,fill=verylightblue,myshadow] (b\i) at (\UVsep,\voff-\i) {$v_{\i}$};
            \fi
        \fi
    }
    
    \foreach \i in {1,...,\Km} {
        \foreach \j in {1,...,\Kn} {
        \ifnum\i=\Kmskip
        \else
            \ifnum\j=\Knskip
            \else
                \draw[-,draw=gray,line width=0.8,line cap=round] ($(a\i.east)$) -- ($(b\j.west)$);
            \fi
        \fi
        }
    }

    \node[draw=none] at ($(a1)!0.5!(b1) + (0,0.5)$) {$K_{m,n}$};

\end{tikzpicture}
